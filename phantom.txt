var page = require('webpage').create();
page.viewportSize={
	width: 800,
	height:1100
};
page.zoomFactor = 1.5;
page.open('http://localhost:3000/file/DGMH.htm', function(status) {
  console.log("Status: " + status);
  if(status === "success") {
    page.render('example.pdf');
  }
  phantom.exit();
});
